def trackback_validate(ignore_unknown_element, tags, uri)
  return if tags.nil?
  if tags.find {|tag| tag == "about"} and
      !tags.find {|tag| tag == "ping"}
    raise MissingTagError.new("#{TRACKBACK_PREFIX}:ping", tag_name)
  end
end
