def xml_stylesheet_pi
  xsss = @xml_stylesheets.collect do |xss|
    pi = xss.to_s
    pi = nil if /\A\s*\z/ =~ pi
    pi
  end.compact
  xsss.push("") unless xsss.empty?
  xsss.join("\n")
end
