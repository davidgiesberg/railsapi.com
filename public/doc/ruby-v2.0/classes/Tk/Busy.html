def method_missing(id, *args)
  name = id.id2name
  case args.length
  when 1
    if name[-1] == =
      configure name[0..-2], args[0]
      args[0]
    else
      configure name, args[0]
      self
    end
  when 0
    begin
      cget(name)
    rescue
      super(id, *args)
    end
  else
    super(id, *args)
  end
end
