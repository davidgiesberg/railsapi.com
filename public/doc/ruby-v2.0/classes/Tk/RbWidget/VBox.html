def _place_config(widget, idx, cnt)
  widget.place_in(self,
                  'relx'=>0, 'x'=>@padx,
                  'rely'=>idx/cnt, 'y'=>@pady,
                  'relwidth'=>1.0, 'width'=>-2*@padx,
                  'relheight'=>1.0/cnt, 'height'=>-2*@pady)
end

def _set_framesize
  bd = self.borderwidth
  self.width(@max_width + 2*@padx + 2*bd)
  self.height((@max_height + 2*@pady) * @widgets.size + 2*bd)
end
