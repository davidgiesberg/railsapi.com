def _from_idx(idx)
  return idx unless idx.kind_of?(String)

  if idx[0] == @  # '@x,y'
    idx
  elsif idx =~ /([^,]+),([^,]+)/
    row = $1, column = $2
    [num_or_str(row), num_or_str(column)]
  else
    num_or_str(idx)
  end
end

def _to_idx(idx)
  if idx.kind_of?(Array)
    idx.collect{|elem| _get_eval_string(elem)}.join(',')
  else
    idx
  end
end
