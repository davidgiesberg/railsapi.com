def initialize(interp, dat)
  # auto-generate tcl/tk representation (variable name)
  exp = TclTk._newname("v_")
  # initialize TclTkObject
  super(interp._tcltkip(), exp)
  # safe this for `set' command
  @set = interp.commands()["set"]
  # set value
  set(dat) if dat
end
