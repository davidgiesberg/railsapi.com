def _warn_
  loc, = caller_locations(2, 1)
  loc = "#{loc.to_s}: " if loc
  warn "#{loc}Use RbConfig instead of obsolete and deprecated Config."
  self
end

def method_missing(*args, &block)
  _warn_
  rbconfig = ::RbConfig
  result = rbconfig.__send__(*args, &block)
  result = rbconfig if rbconfig.equal?(result)
  result
end
